第一个作业

要求：假定给定两个字符串（可以是空串）s1和s2,要将这两个串合并在一起，要求是：s1串的第一个字符与s2串的第一个字符连接在一起，后续以此类推。若一个串的长度大于另一个串的长度，则保留长串的剩余字符，即连接到新串的末尾即可。例如：若s1=’abcd’、s2=’efghi’，则新串为：‘aebfcgdhi’.

s1 = input("请输入一个字符串")
s2 = input("请输入另一个字符串")
i = 0
if(len(s1)>=len(s2)):

    {
    print(s1[i],end=""):
    print(s2[i],end="")
    }
for i in range(len(s1)-len(s2)):#输出更长的那个多余的部分
    {
        print(s1[len(s2)],end="")
    }

for i in range(len(s1)):

    }
for i in range(len(s2)-len(s1)):
    {

第三个作业

要求：

制作随机验证码，不区分大小写。流程：

1）用户执行程序

2）给用户显示需要输入验证码

3）用户输入的值用户输入的值和显示的值相同时显示正确的信息:否则继续生成验证码等待用户输入

import random
while True:
num = []
for i in range(4):
    choose = random.randint(0, 1)
    if choose == 0:
        temp = random.randint(97, 122)
        num.append(chr(temp))  # 生成小写字母
    else:
        temp = random.randint(65, 90)
        num.append(chr(temp))#生成大写字母
print("请输入验证码：%s" % num)
num = input()
index = 0
for i in range(4):
    if num[i].upper() == num[i].upper():
        index += 1
if index != 4:
    print("登录失败")
else:
    print("登录成功")
    break

学生随机分配导师

代码：

import random
from openpyxl import load_workbook
#打开xlsx文件
teachers = []
stu_tea = []
num = []
xlsx_file1 = "待分配导师模板.xlsx"
xlsx_file2 = '待分配学生模板.xlsx'
xlsx_file3 = '随机分配结果.xlsx'
b1 = load_workbook(xlsx_file1)
b2 = load_workbook(xlsx_file2)
b3 = load_workbook(xlsx_file3)
#读取老师数据 以及 最大人数数据 分别存在列表 teachers 和 num
for table1 in b1:
for row in range(2, 9):
    Data1 = table1.cell(row = row , column = 1).value
    teachers.append(Data1)
    Data2 = table1.cell(row = row , column = 2).value
    num.append(Data2)
#将学生信息保存在表中
sheet2 = b2.worksheets[0]
sheet3 = b3.worksheets[0]
for row in range(2, 37):
sheet3.cell(row, 1 , sheet2.cell(row=row, column=2).value)
sheet3.cell(row, 2 , sheet2.cell(row=row, column=1).value)
sheet3.cell(row, 3 , sheet2.cell(row=row, column=3).value)
b3.save(filename = xlsx_file3)
#随机分配导师
j = 0
for i in range(7):
while True:
    if j < num[i] :
        stu_tea.append(teachers[i])
        j = j +1
        continue
    else :
        break
j = 0
random.shuffle(stu_tea)
#将导师姓名存入表格
for i in range(2,37):
sheet3.cell(i , 4, stu_tea[i-1])
b3.save(filename = xlsx_file3)
