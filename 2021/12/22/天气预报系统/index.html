def get_response(url):

#利用request模块进行访问获取内容
response = requests.get(url, headers=headers)
response.encoding = 'utf-8'
# print(response.text)
#利用etree模块，解析返回的内容
selector = etree.HTML(response.text)
#返回etree对象
return selector
