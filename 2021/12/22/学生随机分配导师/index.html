学生随机分配导师

代码：

import random
from openpyxl import load_workbook
#打开xlsx文件
teachers = []
stu_tea = []
num = []
xlsx_file1 = "待分配导师模板.xlsx"
xlsx_file2 = '待分配学生模板.xlsx'
xlsx_file3 = '随机分配结果.xlsx'
b1 = load_workbook(xlsx_file1)
b2 = load_workbook(xlsx_file2)
b3 = load_workbook(xlsx_file3)
#读取老师数据 以及 最大人数数据 分别存在列表 teachers 和 num
for table1 in b1:
for row in range(2, 9):
    Data1 = table1.cell(row = row , column = 1).value
    teachers.append(Data1)
    Data2 = table1.cell(row = row , column = 2).value
    num.append(Data2)
#将学生信息保存在表中
sheet2 = b2.worksheets[0]
sheet3 = b3.worksheets[0]
for row in range(2, 37):
sheet3.cell(row, 1 , sheet2.cell(row=row, column=2).value)
sheet3.cell(row, 2 , sheet2.cell(row=row, column=1).value)
sheet3.cell(row, 3 , sheet2.cell(row=row, column=3).value)
b3.save(filename = xlsx_file3)
#随机分配导师
j = 0
for i in range(7):
while True:
    if j < num[i] :
        stu_tea.append(teachers[i])
        j = j +1
        continue
    else :
        break
j = 0
random.shuffle(stu_tea)
#将导师姓名存入表格
for i in range(2,37):
sheet3.cell(i , 4, stu_tea[i-1])
b3.save(filename = xlsx_file3)
